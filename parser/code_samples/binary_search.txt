func binarySearch(arr: list[int], low: int, high: int, x: int):

    loop while(low <= high):

        mid = low + (high - low) // 2;

        if arr[mid] == x:
            return mid;

        elif arr[mid] < x:
            low = mid + 1;

        else:
            high = mid - 1;

    return -1;
